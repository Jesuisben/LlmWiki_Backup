- 코드 내용
```python
import seaborn as sns
import pandas as pd
import matplotlib.pyplot as plt

from sklearn.model_selection import train_test_split
from sklearn.neighbors import KNeighborsClassifier
from sklearn.metrics import confusion_matrix

plt.rc('font', family='Malgun Gothic')

dataIn, dataOut = '../dataIn/', '../dataOut/'

filename = dataIn + 'likelyhood.csv'

myframe = pd.read_csv(filename)

print(myframe.columns)

# DataFrame에 해당 열의 데이터들을 가져옴
x = myframe [['muscle', 'height', 'grade', 'speech', 'book', 'travel', 'skin']]
# DataFrame에 해당 행의 데이터들을 가져옴
print(x[:10])

y = myframe[['likelyhood']]
print(y[:10])

# value_counts() : DataFrame에 중복되는 value들이 몇개인지 파악
# index에 해당 value의 label / value에 정수 개수를 가진 Series로 반환
print(y.value_counts())

# describe() : DataFrame의 각 숫자형 열의 기초 통계값을 한 번에 계산해서 출력
print("\nbefore min-max normalization")
print(x.describe())

# Min-Max 알고리즘 : 참고 sklearn에 MinMaxScaler 라는 클래스가 있음
# 나중에 이 함수를 여러번 쓸 것 같으면 모듈화를 시켜야 함
def min_max_normalization(data):
    normalized = []

    for value in data :
        normalized_number = (value - min(data))/(max(data) - min(data))
        normalized.append(normalized_number)

    # end for
    return normalized
#end def

x = x.astype(float)

for col in x.columns:
    x.loc[:, col] = min_max_normalization(x[col])
#end for

print("after min-max normalization")
print(x.describe())

# 일관적인 테스트를 위해서 random_state의 값을 일관되게 적어야 함
x_train, x_test, y_train, y_test = train_test_split(x, y, test_size=0.2, random_state=100)

print(f"훈련용 데이터 개수 : {len(x_train)}")
print(f"훈련용 데이터 정답 개수 : {len(y_train)}")

print(f"테스트용 데이터 개수 : {len(x_test)}")
print(f"테스트용 데이터 정답 개수 : {len(y_test)}")

neighbor_count = 3
model = KNeighborsClassifier(n_neighbors=neighbor_count)

y_train = y_train.values.ravel() # 배열 1차원화
model.fit(x_train, y_train)

score = model.score(x_test, y_test)
print(f"scores : {100.0 * score:.2f}")

# 여러 개의 K의 값을 가지고 테스트 하기
k_list = range(1, 10, 2)
accuracies = [] # 정확도를 담기 위한 리스트

for k in k_list:
    model = KNeighborsClassifier(n_neighbors=k)
    model. fit(x_train, y_train)
    myscore = model.score(x_test, y_test)
    accuracies.append(100.0 * myscore)
# end for

# 가장 큰 정확도
max_accuracy = max(accuracies)

# 가장 큰 정확도의 인덱스
max_index = accuracies.index(max_accuracy)

# 해당 K값
best_k = list(k_list)[max_index]

print(f"최고 정확도 : {max_accuracy:.2f}%")
print(f"최적의 K값 : {best_k}")

# plt.plot(k_list, accuracies)
plt.bar(k_list, accuracies)
plt.xlabel('K-value', size = 12)
plt.ylabel('Test Accuracy', size = 12)
plt.title('Likelyhood model Accuracy', size = 18)
plt.savefig(dataOut + 'likelyhood_01.png')


print('최적의 K에 대한 모델을 다시 생성합니다.')
newmodel = KNeighborsClassifier(n_neighbors=best_k)
newmodel.fit(x_train, y_train)

prediction = newmodel.predict(x_test)

cm = confusion_matrix(y_test, prediction)
print('cm')
print(cm)

cm_frame = pd.DataFrame(cm)
sns.heatmap(cm_frame, annot=True, cmap='YlGnBu', fmt='g')
plt.savefig(dataOut + 'likelyhood_02.png')
```

- 코드 내용
```python
import pandas as pd
import matplotlib.pyplot as plt

plt.rc('font', family='Malgun Gothic')

from sklearn.preprocessing import LabelEncoder
from sklearn.model_selection import StratifiedKFold
from tensorflow.python.keras.models import Sequential
from tensorflow.python.keras.layers.core import Dense

dataIn, dataOut = '../dataIn/', '../dataOut/'

# 소나(Sonar) 신호 : 음파를 이용한 물체 탐지 기술(SOund NAvigation Ranging)
# 잠수함의 소나(Sonar) 신호를 이용하여 바위인지(Rock) 기뢰인지(Mine)를 분류하는 이진 분류(Binary Classification) 데이터
# 이미 표준화로 만들어진 데이터입니다.
filename = dataIn + 'sonarTest.csv'

# header=None : 행 머리글이 없습니다.
sonar_frame = pd.read_csv(filename, header=None)
print(sonar_frame.head())

print('\nlabel이 취할수 있는 값 : ', end = '')
print(sonar_frame[60].unique())

print('\n데이터 프레임 정보 확인')
print(sonar_frame.info())

data = sonar_frame.values

print(f'데이터 형상 : {data.shape}')

table_column = data.shape[1] # 컬럼 개수
y_column = 1
x_column = table_column - y_column

x = data[:, 0:x_column].astype(float)
y_raw = data[:, x_column:].ravel() # 1차원화, ['R' 'M'] 중에서 하나임
# print(y_raw)

encoder = LabelEncoder()
encoder.fit(y_raw)
y = encoder.transform(y_raw).astype(float)

n_fold = 10 # 교차할 겹의 개수
seed = 0 # 랜덤 시드 값

skf = StratifiedKFold(n_splits=n_fold, shuffle=True, random_state=seed)
print(f'타입 : {type(skf.split(x, y))}')
print(f'겹 갯수 : {skf.get_n_splits(x, y)}')

cost = [] # for loss value(loss function = cost function_손실 함수)
accuracy = [] # for accuracy(정확도)

print('반복문을 실행 합니다.')
cnt = 0 # 카운터 변수

for train, test in skf.split(x, y):
    cnt += 1
    print(f'{cnt}번째 실행중입니다.')

    model = Sequential()
    model.add(Dense(units=24, input_dim=x_column, activation='relu'))
    model.add(Dense(units=10, activation='relu'))
    model.add(Dense(units=y_column, activation='sigmoid'))

    # cf) 다항 분류 : categorical_crossentropy
    # metrics : 평가할 지표(loss 값은 무조건 출력됨)
    model.compile(loss='binary_crossentropy', optimizer='adam', metrics=['accuracy'])

    model.fit(x[train], y[train], epochs=200, batch_size=5, verbose=0)

    # score는 loss와 accuracy를 저장하고 있는 list 타입입니다.
    score = model.evaluate(x[test], y[test], verbose=0)

    cost.append(score[0])
    accuracy.append(score[1])
# end for

print('\n결과 출력')
avg_cost = sum(cost) / n_fold
avg_accuracy = sum(accuracy) / n_fold

print('=' * 50)
print('교차 검증 결과')
print('=' * 50)

print(f'손실 함수 : {cost}')
print(f'평균 손실 함수 : {avg_cost:.3f}')
print('-' * 50)

print(f'정확도 : {accuracy}')
print(f'평균 정확도 : {avg_accuracy:.3f}')
print('=' * 50)

print('\n그래프 출력')
folds = list(range(1, n_fold + 1))

def draw_line_graph(x, y, xlabel, ylabel, title, filename, marker='o'):
    plt.figure(figsize=(8, 6))
    plt.plot(x, y, marker=marker, linewidth = 2)

    plt.xlabel(xlabel, size=12)
    plt.ylabel(ylabel, size=12)
    plt.title(title, size=18)

    plt.grid(True)
    plt.savefig(dataOut + filename)
    plt.close()
# end def

print('Fold 별 Loss 그래프' )
draw_line_graph(
    folds,
    cost,
    'Fold',
    'Loss',
    'Loss for Each Fold',
    'image01.png'
)

print('Fold 별 Accuracy 그래프' )
draw_line_graph(
    folds,
    accuracy,
    'Fold',
    'Accuracy',
    'Accuracy for Each Fold',
    'image02.png'
)

def draw_bar_graph(x, y, xlabel, ylabel, title, filename):
    plt.figure(figsize=(8, 6))
    plt.bar(x, y)

    plt.xlabel(xlabel, size=12)
    plt.ylabel(ylabel, size=12)
    plt.title(title, size=18)

    plt.grid(axis='y')
    plt.savefig(dataOut + filename)
    plt.close()
# end def

print('Accuracy 막대 그래프')
draw_bar_graph(
    folds,
    accuracy,
    'Fold',
    'Accuracy',
    'Accuracy for Each Fold',
    'image03.png'
)

print('Loss + Accuracy')
plt.figure(figsize=(10, 6))

plt.plot(folds, cost, marker='o', linewidth=2, label='Loss')
plt.plot(folds, accuracy, marker='s', linewidth=2, label='Accuracy')

plt.xlabel('Fold', size=12)
plt.ylabel('Value', size=12)
plt.title('Loss and Accuracy', size=18)
plt.xticks(folds)
plt.xlim([0, 11])

plt.legend()
plt.grid(True)

plt.savefig(dataOut + 'image04.png')
plt.close()
print('Loss + Accuracy')

fig, ax1 = plt.subplots(figsize=(10, 6))

# -------------------------
# 왼쪽 축 : Loss
# -------------------------
ax1.plot(
    folds,
    cost,
    marker='o',
    linewidth=2,
    color='blue',
    label='Loss'
)

ax1.set_xlabel('Fold', fontsize=12)
ax1.set_ylabel('Loss', fontsize=12)

ax1.set_xticks(folds)
ax1.grid(True)

# -------------------------
# 오른쪽 축 : Accuracy
# -------------------------
ax2 = ax1.twinx()

ax2.plot(
    folds,
    accuracy,
    marker='s',
    linewidth=2,
    color='red',
    label='Accuracy'
)
ax2.set_ylabel('Accuracy', fontsize=12)

# -------------------------
# 제목
# -------------------------
plt.title('Loss and Accuracy', fontsize=18)

# -------------------------
# 범례
# -------------------------
lines1, labels1 = ax1.get_legend_handles_labels()
lines2, labels2 = ax2.get_legend_handles_labels()

ax1.legend(
    lines1 + lines2,
    labels1 + labels2,
    loc ='best'
)
plt.savefig(dataOut + 'image05.png')
plt.close()
```
