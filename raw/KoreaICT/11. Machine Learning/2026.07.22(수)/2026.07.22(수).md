- 코드 내용
```python
import matplotlib.pyplot as plt

from keras import Sequential
from keras.layers import Dense, Flatten
from keras.datasets import fashion_mnist

from Utility.keras_graph_util import plot_gray_image, result_dataframe, make_graph_probability

dataOut = "../dataOut/"

# load.data() 함수가 어떻게 만들어졌는지 Ctrl + 클릭으로 들어가서 return 값을 보고
# 그 값을 그대로 이용해서 변수 정의 및 할당 시킴
(x_train, y_train), (x_test, y_test) = fashion_mnist.load_data()

class_names = ['T-shirt/top', 'Trouser', 'Pullover', 'Dress', 'Coat', 'Sandal', 'Shirt', 'Sneaker', 'Bag', 'Ankle boot']

idx = y_train[0]
print(f"# 0번째 이미지의 label(정답) : {idx}")
print(f"# class 이름 : {class_names[idx]}")

plt.figure()
plt.imshow(x_train[0])
plt.colorbar()
plt.grid(False)
filename = "fashion_mnist_01.png"
plt.savefig(dataOut + filename)
plt.close()
print(f"{filename} 파일 저장 완료")

x_train = x_train / 255.0
x_test = x_test / 255.0

num_rows, num_cols = 4, 4

plot_gray_image(x_train, y_train, num_rows, num_cols, "fashion_mnist_02.png", class_names)

NB_CLASSES = len(class_names)

model = Sequential([
    Flatten(input_shape=(28, 28)),
    Dense(units=128, activation="relu"),
    Dense(units=128, activation="relu"),
    Dense(units=NB_CLASSES, activation="softmax")
])

# sparse_matrix
model.compile(loss="sparse_categorical_crossentropy", optimizer="adam", metrics=["accuracy"])

model.fit(x_train, y_train, epochs=5, verbose=0)

score_list = model.evaluate(x_test, y_test, verbose=0)

print(f"손실 함수={score_list[0]}")
print(f"정확도={score_list[1]}")

prediction = model.predict(x_test)

result_dataframe(prediction, y_test, "dataframe.csv")

make_graph_probability(nrow=num_rows, ncol=num_cols, prediction=prediction, true_label=y_test,
                       image=x_test, class_names=class_names, file_name='fashion_mnist_03.png')

print("finished")
```
