```python
import matplotlib.pyplot as plt

from keras import Sequential
from keras.layers import Dense, Flatten
from keras.datasets import fashion_mnist

from Utility.keras_graph_util import plot_gray_image, result_dataframe, make_graph_probability

dataOut = "../dataOut/"

# load.data() 함수가 어떻게 만들어졌는지 Ctrl + 클릭으로 들어가서 return 값을 보고
# 그 값을 그대로 이용해서 변수 정의 및 할당 시킴
(x_train, y_train), (x_test, y_test) = fashion_mnist.load_data()

class_names = ['T-shirt/top', 'Trouser', 'Pullover', 'Dress', 'Coat', 'Sandal', 'Shirt', 'Sneaker', 'Bag', 'Ankle boot']

idx = y_train[0]
print(f"# 0번째 이미지의 label(정답) : {idx}")
print(f"# class 이름 : {class_names[idx]}")

plt.figure()
plt.imshow(x_train[0])
plt.colorbar()
plt.grid(False)
filename = "fashion_mnist_01.png"
plt.savefig(dataOut + filename)
plt.close()
print(f"{filename} 파일 저장 완료")

x_train = x_train / 255.0
x_test = x_test / 255.0

num_rows, num_cols = 4, 4

plot_gray_image(x_train, y_train, num_rows, num_cols, "fashion_mnist_02.png", class_names)

NB_CLASSES = len(class_names)

model = Sequential([
    Flatten(input_shape=(28, 28)),
    Dense(units=128, activation="relu"),
    Dense(units=128, activation="relu"),
    Dense(units=NB_CLASSES, activation="softmax")
])

# sparse_matrix
model.compile(loss="sparse_categorical_crossentropy", optimizer="adam", metrics=["accuracy"])

model.fit(x_train, y_train, epochs=5, verbose=0)

score_list = model.evaluate(x_test, y_test, verbose=0)

print(f"손실 함수={score_list[0]}")
print(f"정확도={score_list[1]}")

prediction = model.predict(x_test)

result_dataframe(prediction, y_test, "dataframe.csv")

make_graph_probability(nrow=num_rows, ncol=num_cols, prediction=prediction, true_label=y_test,
                       image=x_test, class_names=class_names, file_name='fashion_mnist_03.png')

print("finished")
```

import seaborn as sns
import pandas as pd
import matplotlib.pyplot as plt

plt.rc('font', family='Malgun Gothic')
plt.rcParams['axes.unicode_minus'] = False

from sklearn.model_selection import train_test_split
from sklearn.tree import DecisionTreeClassifier
from sklearn.tree import plot_tree
from sklearn.metrics import confusion_matrix
from sklearn.tree import export_text

dataIn, dataOut = '../dataIn/', '../dataOut/'

df = pd.read_csv(dataIn + 'Europe Hotel Booking Satisfaction Score.csv')
print("df.columns")
print(df.columns)

# 결측치 확인
print("\ndf.isnull().sum()")
print(df.isnull().sum())

wifi = "Hotel wifi service"
print(df[wifi].unique())

ordering = df[wifi].value_counts().index

sns.countplot(df, x=wifi, order=ordering)
plt.savefig(dataOut + wifi + ".png")
plt.close()

plt.figure(figsize=(15, 10))
df_corr = df.iloc[:, 6:].corr(numeric_only=True)
sns.heatmap(df_corr, annot=True, cmap="Blues")
plt.savefig(dataOut + "correlation_heatmap.png")
plt.close()

x = df.iloc[:, 6:16]
y = df.iloc[:, -1] # 맨 뒤 컬럼 1개

train_features = x.columns
print('train_features')
print(train_features)

x_train, x_test, y_train, y_test = train_test_split(x, y, test_size=0.3, random_state=42, stratify=y)

model = DecisionTreeClassifier()
model.fit(x_train, y_train)

TREE_MAX_DEPTH = 3

plt.figure(figsize=(15, 10))
plot_tree(model, feature_names=train_features, max_depth=TREE_MAX_DEPTH, filled=True, rounded=True)
plt.savefig(dataOut + "plot_tree.png")
plt.close()

most_importances = model.feature_importances_
ft_frame = pd.DataFrame(most_importances, index=train_features, columns=["Importance"])
ft_frame = ft_frame.sort_values(by="Importance", ascending=False)
print(ft_frame)

max_index = ft_frame['Importance'].idxmax ()
print(f'가장 큰 값의 색인 : {max_index}')

min_index = ft_frame['Importance'].idxmin()
print(f'가장 작은 값의 색인 : {min_index}')

plt.figure(figsize=(15, 10))
sns.barplot(x=ft_frame.Importance, y=ft_frame.index, hue=ft_frame.index, legend=False, palette="viridis")
plt.title("특성 중요도 시각화", size = 18)
plt.xlabel("중요도")
plt.ylabel("특성(Feature)")
plt.savefig(dataOut + "n_features.png")
plt.close()

print(f'그림을 보면 호텔 만족도에 가장 큰 영향을 주는 특성은 "{max_index}"입니다.')

print("\n훈련 데이터를 이용한 모델 분류 정확도 : ", model.score(x_train, y_train))

# [4] 모델 평가 및 예측하기
# 모델 평가하기
print("테스트 데이터 성능평가 : ", model.score(x_test, y_test))

#  테스트 데이터 예측하기
prediction = model.predict(x_test)
print(prediction[:5])
print(y_test[:5])

prediction = model.predict(x_test)
conf_matrix = confusion_matrix(y_test, prediction)

plt.figure(figsize=(8, 6))
sns.heatmap(conf_matrix, annot=True, cmap='Blues', fmt='d')
plt.title('Hotel satisfaction classification')
plt.xlabel('Predicted')
plt.ylabel('Actual')
plt.savefig(dataOut + 'confusion_matrix.png')

df_new = pd.read_csv(dataIn + 'hotel_satisfaction_new.csv')

print(model.predict(df_new))

tree_rules = export_text(model, feature_names=train_features, max_depth=TREE_MAX_DEPTH)
print(tree_rules)

filename = 'tree_rules.txt'
with open(dataIn + filename, mode='wt', encoding='UTF-8') as rulefile:
    rulefile.write(tree_rules)

print(f"{filename} 파일이 저장되었습니다.")
