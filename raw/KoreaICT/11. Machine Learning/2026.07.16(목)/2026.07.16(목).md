```python
##################################################################
import sys, math
import pandas as pd

from konlpy.tag import Komoran


import os
javahome = 'JAVA_HOME'
os.environ[javahome] = r'C:\Users\ICT02-006\.jdks\ms-21.0.10\bin\server'

# print(javahome in os.environ)

dataIn = "../dataIn/"
##################################################################
# 클래스 생성
##################################################################
class BayesianFilter : # 베이지안 필터
    #---------------------------------------------------------------
    # 생성자
    #---------------------------------------------------------------
    def __init__(self):
        self.words = set() # 단어 집합
        self.word_dict = {} # 카테고리별 단어 빈도 저장
        self.category_dict = {} # 카테고리별 문서 개수 저장

        try:
            self.komoran = Komoran(userdic=dataIn + "user_dic.txt")
            print("사용자 정의 사전 적용")

        except Exception as err :
            self.komoran = Komoran()
            print("기본 사전 사용")
            print(err)

        print("생성자 호출됨")
    #---------------------------------------------------------------
    # 학습을 위한 함수
    #---------------------------------------------------------------
    def fit(self, sentense, category):
        # print(f"문장 : {sentense}, 카테고리 : {category}")
        # 형태소 분석, 분리
        word_list = self.bayes_split(sentense)
        print(word_list)

        for word in word_list: # ["세일", "무료", "배송", "할인"] 중 하나가 word가 됩니다.
            self.increase_word(word, category)

        self.increase_category(category)

        print("\n[학습 결과 출력]")
        print(f"문장 : {sentense}")
        print(f"카테고리 : {category}")

        print("\nword_dict")
        print(self.word_dict)

        print("\nwords")
        print(self.words)

        print("\ncategory_dict")
        print(self.category_dict)

        print("\n" + "=" * 50)
    #---------------------------------------------------------------
    # 해당 메일 제목을 이용하여 형태소 분석
    #---------------------------------------------------------------
    def bayes_split(self, phrase):
        malist = self.komoran.pos(phrase)
        # print(f"\n단어 : {phrase}")
        # print(f"형태소 분석 : {malist}")

        result_list = []

        # 품사 태깅 (POS tagging)
        for word, pos in malist : # tagging 공부 (각 단어에 품사 정보를 붙이는 것)
            if pos in ["NNG", "NNP", "SL"] : # 일반 명사, 고유 명사, 영문
                result_list.append(word)

        return result_list
    #---------------------------------------------------------------
    # 단어의 집합(words)과 카테고리별 단어 빈도 사전(word_dict)에 데이터 저장하기
    #---------------------------------------------------------------
    def increase_word(self, word, category):
        if category not in self.word_dict:
            self.word_dict[category] = {}

        # 신규 단어를 사전에 새롭게 추가
        if word not in self.word_dict[category]:
            self.word_dict[category][word] = 0

        # 이전, 신규 상광 없이 카운터 1증가
        self.word_dict[category][word] += 1

        self.words.add(word) # 사전에 단어 추가
    #---------------------------------------------------------------
    # 카테고리 사전에서 해당 카테
    #---------------------------------------------------------------
    def increase_category(self, category):
        # 신규 카테고리에 대한 value를 0으로 추가합니다.
        if category not in self.category_dict:
            self.category_dict[category] = 0

        # 해당 카테고리의 합을 1증가 시킵니다.
        self.category_dict[category] += 1
    #---------------------------------------------------------------
    # 테스트용 메일 email을 가지고, 예측을 진행합니다.
    #---------------------------------------------------------------
    def predict(self, email):
        # words : 형태소 분석이 완료된 단어 목록
        words = self.bayes_split(email)

        answer_category = None # 정답 카테고리

        # 카테고리별 점수 튜플을 담을 리스트
        score_list = [] # 예시 : [("광고", 점수1), ("일반", 점수2)]

        # max_score : 최고 점수
        max_score = -sys.maxsize # 시스템이 가지고 있는 제일 작은 정수를 기본 값

        # 카테고리별 점수 계산하기
        for category in self.category_dict:
            score = self.score(words, category)

            # 해당 카테고리의 점수를 list에 추가하기
            score_list.append((category, score))
            print(f"'{category}' 점수 = {score:.4f}")

            if score > max_score :
                max_score = score # 최고 점수 교체
                answer_category = category # 정답 카테고리 갱신
            # end if
        # end for

        return answer_category, score_list
    # ---------------------------------------------------------------
    # 형태소 목록과 카테고리 정보를 사용하여 점수를 계산합니다.
    # ---------------------------------------------------------------
    def score(self, words, category):
        # underflow를 방지하기 위하여 log 함수를 사용합니다.
        score = math.log(
            self.category_probability(category)
        )

        for word in words :
            score += math.log(
                self.word_probability(word, category)
            )

        return score
    # ---------------------------------------------------------------
    # 카테고리 확률 = (해당 카테고리/전체)을 구합니다.
    # ---------------------------------------------------------------
    def category_probability(self, category):
        total = sum(self.category_dict.values())
        return self.category_dict[category] / total
    # ---------------------------------------------------------------
    # 카테고리별 해당 단어에 대한 확률 정보를 구합니다.
    # 라플라스 스무딩을 적용합니다.
    # ---------------------------------------------------------------
    def word_probability(self, word, category):
        # 해당 카테고리에 들어 있는 워드의 수를 반환받습니다.
        # 라플라스 스무딩을 위하여 1을 더해 줍니다.
        if word in self.word_dict[category]:
            word_count = self.word_dict[category][word] + 1
        else :
            word_count = 1
        # end if

        total = sum(self.word_dict[category].values())
        total += len(self.words)

        return word_count / total
##################################################################
# 메인 로직
##################################################################
mail_data = pd.read_csv(dataIn + "mail_data.csv")
bayes_filter = BayesianFilter()

print("\n학습 시작")
# iterrows() : DataFrame을 행 단위로 반복하며, (행 인덱스, 행 데이터) 형태의 튜플을 반환한다.
# 튜플의 첫 번째 요소는 행 인덱스이고, 두 번째 요소는 Series 형태의 행 데이터이다.
# 다음 반복문에서는 행 인덱스가 _에 저장되고, 행 데이터가 mail에 저장된다.
for _, mail in mail_data.iterrows():
    bayes_filter.fit(mail["text"], mail["category"])
print("학습 종료")

print(f"\n등록된 카테고리 : {bayes_filter.category_dict}")

print(f"\n전체 단어 수 : {len(bayes_filter.words)}")
##################################################################

##################################################################
# 테스트하기
##################################################################
test_data = [

]

print("\n단어별 테스트 결과 출력하기")
for text in test_data:
    print("입력 :", text)

    category, scores = bayes_filter.predict(text)

    print("예측 결과 :", category)
    print("점수 :", scores)

    print("-"*40)
##################################################################
```

- 구현할 내용
```text
VGG16 모델(사전 학습 모델)을 다운로드하고, 해당 모델에 대한 간략한 정보를 확인합니다.
이미지 파일 목록에 반복문을 사용하여
	품종을 확인할 이미지 목록을 로딩합니다.
	해당 이미지를 다시 파일로 저장해 봅니다.
	해당 이미지를 ndarray 타입으로 변환합니다.
	ndarray 타입을 VGG16 모델에 입력하기 위한 포맷으로 변경합니다.
해당 배열(arr_input)을 모델에 넣어서 예측을 수행해 봅니다.
상위 10개의 예측 결과를 클래스 이름으로 변환합니다.
원본 이미지와 예측 결과 및 확률 값을 데이터 프레임으로 생성 후 csv 파일을 작성해 봅니다.
이미지에 대한 확률 값을 이용하여 막대 그래프를 그려 봅니다.
```

- 코드 내용
```python
import os
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

from keras.applications import VGG16
from keras.applications.vgg16 import preprocess_input, decode_predictions
from keras.utils import load_img, img_to_array


from Utility.keras_graph_util import model_information

plt.rc('font', family='Malgun Gothic')
plt.rcParams['axes.unicode_minus'] = False

# 모델이 사용한 이미지 크기
IMAGE_WIDTH, IMAGE_HEIGHT = 224, 224

image_source= '../dataIn/image/' # 이미지 출처
image_target = '../dataOut/' # 결과가 저장될 폴더

file_list = os.listdir(image_source)

test_images = [] # 예측을 수행하고자 하는 전처리된 이미지 관련 배열들을 저장하고 있는 리스트입니다.

for filename in file_list:
    # 현재 진행 중인 이미지 파일을 VGG16 모델의 이미지 크기로 변경하여 읽어 들입니다.
    curr_image = load_img(image_source + filename, target_size=(IMAGE_WIDTH, IMAGE_HEIGHT))
    plt.imshow(curr_image)

    # 해당 이미지를 다시 파일로 저장해 봅니다.
    saved_file = image_target + 'pretrained_model_' + filename
    plt.savefig(saved_file)
    print(f"{filename} 파일 저장 완료")

    ndarray = img_to_array(curr_image) # Image -> Array
    # print(ndarray)

    # 해당 이미지를 VGG16이 학습했던 환경과 동일하게 형상을 변경해 줍니다.
    pre_input = preprocess_input(ndarray)
    # print(pre_input)

    print(f"이미지 형상 : {pre_input.shape}") # (224, 224, 3)

    test_images.append(pre_input)

    print("-" * 40)
# end for

x_test = np.stack(test_images)
# 4차원으로 만들기 위해서
print(f"최종 입력 이미지의 형상 : {x_test.shape}") # (4, 224, 224, 3)


model = VGG16() # 사전 학습 모델 읽기
print('type(mode1)')
print(type(model))

model_information(model) # 모델 정보 확인

# 사전 학습 모델이므로, 다시 학습할 필요가 없습니다.
# 바로 예측을 진행합니다.
prediction = model.predict(x_test)
print(f'prediction.shape = {prediction.shape}')
print(f'입력 데이터 개수 = {prediction.shape[0]}')
print(f'VGG16 모델의 class 개수 = {prediction.shape[1]}')
print(prediction)


print('예측 값 표시 : ')
probability_topn = decode_predictions(prediction, top=10)
print(probability_topn)

# csv 파일에 저장될 데이터 리스트
csv_list = []

for idx in range(len(file_list)):
    print(f'이미지 : `{file_list[idx]}`의 확률 값')

    for prob in range(len(probability_topn[idx])):
        # sublist에 들어 가는 순서 : (이미지이름, 결과문자열, 확률값)
        sublist = [file_list[idx],
                   probability_topn[idx][prob][1],
                   probability_topn[idx][prob][2]]
        csv_list.append(sublist)
    print('-' * 30)
    # end inner for
# end outer for

print(csv_list)

mycolumns = ['image', 'description', 'probability']
df = pd.DataFrame(csv_list, columns=mycolumns)

dataOut = '../dataOut/'
filename = dataOut + 'prediction_result.csv'
df.to_csv(filename, encoding='UTF-8')
print(f'{filename} 파일 생성됨')

# 이미지별 상위 top 10을 막대 그래프로 시각화.
mycolor = ['b', 'g', 'r', 'c', 'm', 'y', 'k', 'Beige', '#FF1493', '#FF00CC']

for idx, prob_list in enumerate(probability_topn):
    plt.figure(figsize=(10, 8))

    # 리스트 컴프리헨션으로 data와 captions을 생성합니다.
    captions = [item[1] for item in prob_list]
    data = [100 * item[2] for item in prob_list] # 확률값에 100곱하기

    # 시리즈로 데이터 생성 후 차트 그리기
    chartdata = pd.Series(data, index=captions)
    chartdata.plot(kind='bar', rot=12, color=mycolor[:len(captions)])

    # 제목 및 기타 설정
    plt.title(f'이미지 {file_list[idx]} 분류 결과', size=15)
    plt.ylim([-10, 100])
    plt.grid(True)

    filename = f'{image_target}probability_{file_list[idx]}'
    plt.savefig(filename)
    print(f'{filename} 파일이 저장되었습니다.')

# end for

print('finished')
```
