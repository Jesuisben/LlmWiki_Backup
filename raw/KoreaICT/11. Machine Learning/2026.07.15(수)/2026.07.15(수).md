입력 데이터

결과물

- 코드 내용
```python
import codecs  
from bs4 import BeautifulSoup  
from konlpy.tag import Okt  
from gensim.models import word2vec  
  
import os  
javahome = 'JAVA_HOME'  
os.environ[javahome] = r'C:\Users\ICT02-006\.jdks\ms-21.0.10\bin\server'  
  
# print(javahome in os.environ)  
  
# 소설 '토지'를 읽어 와서, 형태소 분석을 수행한 다음 model 파일로 저장해 봅니다.  
  
dataIn, dataOut = '../dataIn/', '../dataOut/'  
# utf-16 인코딩으로 파일을 열고 글자를 출력하기 --- (※1)  
fp = codecs.open(filename= dataIn + 'BEXX0003.txt', mode='r', encoding='UTF-8')  
soup = BeautifulSoup(fp, 'html.parser')  
# body 태그의 바로 하위의 text 선택 (> : 직계 자손 선택자)  
# <text>가 해당 파일에 하나 밖에 없다면 body = soup.select_one('text')과 똑같지만  
# 만약 다른 <text>가 있다면 구조상 더 위에있는 <text>를 가져옴  
# 직계 자손 선택자를 사용하는 이유는 내가 원하는 대상을 정확하게 선택하기 위함임  
body = soup.select_one('body > text')  
# getText()는 get_text()의 별칭(alias)임  
documents = body.getText()  
# print( text )  
  
# 텍스트를 한 줄씩 처리하기 --- (※2)  
okt = Okt()  
  
setenses = documents.split('\r\n')  
# print( lines )  
  
final_sentenses = []  
  
for oneline in setenses:  
    # 형태소 분석하기 --- (※3)  
    # 단어의 기본형 사용  
    # norm=True : 되욬ㅋㅋ -> 되요  
    # stem-True : 동사/형용사를 기본형으로 변환  
    malist = okt.pos(oneline, norm=True, stem=True)  
    # print( malist )  
  
    filtered_words = []  
  
    for word, pos in malist: # word는 단어, pos는 품사  
        # 어미/조사/구두점 등은 대상에서 제외  
        if not [pos] in ["Punctuation", "Josa", "Eomi"]:# 어미  
            filtered_words.append(word)  
  
    # 추가 필터링이 필요하다면 여기서 하면 좋을 듯...  
    #   빈도수 몇개 이하, 글자 길이 몇글자 이하  
  
    join_result = (" ".join(filtered_words)).strip()  
  
    # print(filtered_words)  
    # print(join_result)  
    final_sentenses.append(join_result)  
# end for  
  
# print(final_sentenses)  
  
  
  
# 파일로 출력하기  --- (※4)  
# toji.wakati 파일은 형태소 분석(Morphological Analysis)과 불필요한 품사 제거가 완료된 텍스트 파일입니다.  
tokens_file = 'toji.wakati' # 형태소 분석이 완료되고, 불필요한 품사 제거가 완료된 파일  
  
with open(dataOut + tokens_file, 'wt', encoding='UTF-8') as tojifile:  
    tojifile.write('\n'.join(final_sentenses))  
  
print(f'파일 : {tokens_file} 저장 완료')  
  
# Word2Vec 모델 만들기 --- (※5)  
word_sentenses = word2vec.LineSentence(dataOut + tokens_file)  
  
  
model = word2vec.Word2Vec(  
    word_sentenses,  
    vector_size=200, # 단어 1개를 200차원의 벡터로 표현하겠다. (일반적으로 50~300)    window=10, # 이웃 단어 계수  
    hs=1, # Hierarchical Softmax 사용  
    min_count=2, # 2번 이상 출현한 단어들만 학습을 하겠다.  
    sg=1 # 학습 방식(0:주변 단어를 보고 중심 단어 예측 , 1:0의 반대)  
)  
  
saved_file_name = dataOut + 'toji.model'  
model.save(saved_file_name)  
print(f'파일 : {saved_file_name} 저장 완료')
```
