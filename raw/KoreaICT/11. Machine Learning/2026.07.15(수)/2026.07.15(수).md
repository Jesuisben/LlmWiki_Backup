- 코드 내용
```python
import codecs  
  
from bs4 import BeautifulSoup  
from konlpy.tag import Okt  
from gensim.models import word2vec  
  
import os  
javahome = 'JAVA_HOME'  
# jvm.dll 이 파일이 있는 경로를 찾아야 함  
os.environ[javahome] = r'C:\Users\ICT02-006\.jdks\ms-21.0.10\bin\server'  
  
# print(javahome in os.environ)  
  
# 소설 '토지'를 읽어 와서, 형태소 분석을 수행한 다음 model 파일로 저장해 봅니다.  
dataIn, dataOut = '../dataIn/', '../dataOut/'  
  
# utf-16 인코딩으로 파일을 열고 글자를 출력하기 --- (※1)  
myfile = codecs.open(filename= dataIn + 'BEXX0003.txt', mode='r', encoding='UTF-8')  
  
soup = BeautifulSoup(myfile, 'html.parser')  
  
# body 태그의 바로 하위의 text 선택 (> : 직계 자손 선택자)  
# <text>가 해당 파일에 하나 밖에 없다면 body = soup.select_one('text')과 똑같지만  
# 만약 다른 <text>가 있다면 구조상 더 위에있는 <text>를 가져옴  
# 직계 자손 선택자를 사용하는 이유는 내가 원하는 대상을 정확하게 선택하기 위함임  
body = soup.select_one('body > text')  
  
# getText()는 get_text()의 별칭(alias)임  
documents = body.getText()  
# print('documents')  
# print(documents)  
  
  
# 텍스트를 한 줄씩 처리하기 --- (※2)  
okt = Okt()  
  
sentenses = documents.split('\r\n')  
# print(len(sentenses))  
# print(sentenses)  
  
final_sentenses = []  
  
for oneline in sentenses:  
    # 형태소 분석하기 --- (※3)  
    # 단어의 기본형 사용  
    # norm=True : 되욬ㅋㅋ -> 되요  
    # stem-True : 동사/형용사를 기본형으로 변환  
    malist = okt.pos(oneline, norm=True, stem=True)  
    # print(malist)  
  
    filtered_words = []  
  
    for word, pos in malist: # word는 단어, pos는 품사  
        # 어미/조사/구두점 등은 대상에서 제외  
        if not pos in ['Punctuation', 'Josa', 'Eomi']:  # 어미  
            filtered_words.append(word)  
  
    # 추가 필터링이 필요하다면 여기서 하면 좋을 듯...  
    #   빈도수 몇개 이하, 글자 길이 몇글자 이하  
  
    join_result = (" ".join(filtered_words)).strip()  
  
    # print(filtered_words)  
    # print(join_result)  
    final_sentenses.append(join_result)  
# end for  
  
# print(final_sentenses)  
  
# 파일로 출력하기  --- (※4)  
# toji.wakati 파일은 형태소 분석(Morphological Analysis)과 불필요한 품사 제거가 완료된 텍스트 파일입니다.  
tokens_file = dataOut + 'toji.tokens' # 형태소 분석이 완료되고, 불필요한 품사 제거가 완료된 파일  
  
with open(tokens_file, mode='wt', encoding='UTF-8') as tojifile:  
    tojifile.write('\n'.join(final_sentenses))  
  
print(f'파일 : {tokens_file} 저장 완료')  
  
# Word2Vec 모델 만들기 --- (※5)  
# LineSentence : 텍스트 파일을 한 줄씩 읽어서 단어 list로 변환해 줍니다.  
word_sentenses = word2vec.LineSentence(tokens_file)  
  
#  Embedding : OpenAi에서 사용하는 용어  
model = word2vec.Word2Vec(  
    word_sentenses,  
    vector_size=200, # 단어 1개를 200차원의 벡터로 표현하겠다. (일반적으로 50~300)    window=10, # 이웃 단어 계수  
    hs=1, # Hierarchical Softmax 사용  
    min_count=2, # 2번 이상 출현한 단어들만 학습을 하겠다.  
    sg=1 # 학습 방식(0:주변 단어를 보고 중심 단어 예측 , 1:0의 반대)  
)  
  
saved_file_name = dataOut + 'toji.model'  
model.save(saved_file_name)  
print(f'파일 : {saved_file_name} 저장 완료')
```

```python
################################################  
import math  
import matplotlib.pyplot as plt  
  
from gensim.models import word2vec  
from wordcloud import WordCloud  
  
plt.rc('font', family='Malgun Gothic')  
plt.rcParams['axes.unicode_minus'] = False  
  
dataOut = '../dataOut/'  
##################################################################  
# 함수 정의  
##################################################################  
def show_similiar(hot_keyword):  
    # wv : Word Vector  
    # 해당 모델에서 hot_keyword와 가장 유사도가 높은 단어들의 유사도를 반환  
    result_words = model.wv.most_similar(  
        positive=[hot_keyword]  
    )  
    return result_words  
  
  
def graph_horizontal_bar(title, draw_data):  
    tick_size = len(draw_data) # 그리고자 하는 전체 데이터 개수  
  
    # x[0]은 사실상 similiar_words[0]    word = [x[0] for x in draw_data] # 축에 보여지는 항목들  
    # x[1]은 사실상 similiar_words[1]    score = [x[1] for x in draw_data] # 그려지는 유사도 데이터  
  
    plt.figure(figsize=(8, 5))  
    plt.title(f"단어 '{title}'과(와)의 유사도")  
    plt.barh(range(tick_size), score, align="center")  
    plt.yticks(range(tick_size), word)  
  
    # X축 눈금 사용자 정의  
    min_value = min(score)  
    max_value = max(score)  
  
    xmin = math.floor(min_value * 10) / 10  
    xman = math.ceil(max_value * 10) / 10  
  
    plt.xlim(xmin, xman) # 상하한값 사용자 조정  
  
    plt.grid(True)  
  
    filename = "word2vec_bar.png"  
    plt.savefig(dataOut + filename)  
    print(f"파일 {filename} 저장 완료")  
  
def draw_word_cloud(word_data):  
    wc = WordCloud(  
  
    ).generate_from_frequencies(word_data)  
  
    plt.figure(figsize=(12, 8))  
    plt.title("토지 Word2Vec 워드 클라우드")  
    plt.imshow(wc)  
  
    word_image = "wordcloud_image.png"  
    plt.savefig(dataOut + word_image)  
    print(f"파일 {word_image} 저장 완료")  
##################################################################  
# Word2Vce 모델 읽기  
##################################################################  
filename = dataOut + 'toji.model'  
model = word2vec.Word2Vec.load(filename)  
print(f'{filename} 파일 로딩 완료')  
##################################################################  
# 유사 단어 검색  
##################################################################  
keyword = '집'  
similiar_words = show_similiar(keyword)  
  
print(f"\n{keyword}와 유사한 데이터")  
for word, score in similiar_words:  
    print(f"단어 : {word}, 유사도 점수 : {score:.4f}")  
  
graph_horizontal_bar(keyword, similiar_words)  
##################################################################  
# Word2Vec Vocabulary 빈도 이용  
##################################################################  
word_dictionary = {}  
  
# print(model.wv.index_to_key) # 학습된 단어 목록  
for word in model.wv.index_to_key :  
    # word : 단어, count : 빈도 수  
    # vecattr = vec + attr  
    count = model.wv.get_vecattr(word, "count")  
  
    # 빈도 수가 60보다 크고, 단어의 길이가 2이상인 항목만 사용할게요.  
    if count > 60 and len(word) >= 2:  
        word_dictionary[word] = count  
  
print(word_dictionary)  
##################################################################  
# 주요 단어 목록  
##################################################################  
print("\n주요 단어 목록")  
  
for word, count in sorted(  
    word_dictionary.items(),  
    key=lambda x: x[1],  
    reverse=True  
) :  
    print(f"{word}({count})")  
##################################################################  
# 워드 클라우드 생성  
##################################################################  
draw_word_cloud(word_dictionary)  
##################################################################
```
