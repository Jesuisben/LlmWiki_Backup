- 코드 내용
```python
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from sklearn.linear_model import LinearRegression
from sklearn.model_selection import train_test_split

plt.rc('font', family='Malgun Gothic')

plt.rcParams['axes.unicode_minus'] = False

pd.set_option('display.max_columns', None)

import seaborn as sns

df = pd.read_csv('../dataIn/multiple_linear_regression_dataset.csv')

print(df.head())
print()
print(df.info())
print()
print(df.describe())
print()
print(df.isnull().sum())
print()
# duplicated()는 True 혹은 False 반환함
print(df.duplicated().sum())

dataOut = "../dataOut/"

sns.pairplot(df)
plt.savefig(dataOut + "pairplot.png")

# correlation (상관계수)
corr = df.corr(numeric_only=True)
print(corr)

mask = np.triu(np.ones_like(corr, dtype=bool))
plt.figure(figsize=(10, 8))
sns.heatmap(corr, mask=mask, cmap='coolwarm', annot=True, fmt='.2f', cbar=True, square=True)
plt.title('변수들간 상관 계수', size=15)
filename = dataOut + 'heatmap.png'
plt.savefig(filename)


xdata = ['age', 'experience']
fig, axes = plt.subplots(1, 2, figsize=(14, 5))

for idx in range(len(xdata)):
    axes[idx].scatter(df[f'{xdata[idx]}'], df['income'], c='coral', s=50)
    axes[idx].set_title(f'{xdata[idx]} vs Income')
    axes[idx].set_xlabel(f'{xdata[idx]}')
    axes[idx].set_ylabel('Income')

filename = dataOut + 'scatter.png'
plt.savefig(filename)
print(filename + ' 파일이 저장되었습니다.')

x = df[["age", "experience"]]
y = df["income"]

# 10세트는 따로 빼둠
x_train, x_test, y_train, y_test = train_test_split(x, y, test_size=0.3, random_state=10)

model = LinearRegression()

model.fit(x_train, y_train)

r_square = model.score(x_test, y_test)
print(f"결정 계수 : {r_square}")
print(f"\n기울기 a : {model.coef_}")
print(f"\ny절편 b : {model.intercept_}")

y_pred = model.predict(x_test)


test_preds = pd.DataFrame(y_test)
test_preds.columns = ["정답"]
# 존재하지 않는 컬럼 이름에 값을 대입하면 새로운 컬럼이 자동으로 추가됨
test_preds["예측"] = y_pred
test_preds["mse"] = (test_preds["예측"] - test_preds["정답"]) ** 2
print("test_preds")
print(test_preds)

filename = dataOut + "test_preds.csv"
test_preds.to_csv(filename, index=False, encoding="UTF-8")

mse = test_preds["mse"].mean()
print(f"평균 제곱 오차 : {mse}")


plt.figure()

sns.regplot(x='정답', y='예측',  data=test_preds, line_kws={"color": "red"});
plt.title('정답과 예측치의 산점도')
plt.xlim([15000, 45000])
plt.ylim([15000, 45000])
plt.tight_layout()
filename = dataOut + 'regplot.png'
plt.savefig(filename)
print(filename + ' 파일이 저장되었습니다.')
```

- 구현할 내용
```text
iris 파일을 읽고, 데이터의 일부를 출력해 봅니다.
각 품종별 데이터의 개수를 파악해 봅니다.
LabelEncoder 클래스를 사용하여 문자열로 되어 있는 품종 컬럼에 대하여 숫자 형태로 변경합니다.
각 품종별로 꽃받침의 길이(Sepal.Length 컬럼)와, 꽃받침의 너비(Sepal.Width 컬럼)에 대하여 산점도 그래프를 그려 봅니다.
입력과 출력 데이터 셋을 이용하여 8대2로 데이터를 분리합니다.
각 특성(컬럼)별로 데이터의 최소와 최대 값을 확인해 봅니다.
StandardScaler 클래스를 사용하여 분리된 데이터에 대하여, 표준 정규화를 수행합니다.
SVC 클래스를 사용하여 SVM 모델을 생성합니다.
해당 모델을 이용하여 학습을 진행하고, 테스트용 데이터를 사용하여 예측해 봅니다.
예측된 값(prediction)과 정답(label) 데이터 및 가채점 결과를 출력해 봅니다.
정답의 개수 및 정확도를 구하여 출력합니다.
모형 성능 평가를 위한 지표를 계산하여, 보고서를 작성합니다.
Confusion Matrix를 히트맵으로 그려 봅니다.
```

- 코드 내용
```python
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import LabelEncoder

plt.rc('font', family='Malgun Gothic')
plt.rcParams['axes.unicode_minus'] = False

dataIn, dataOut = "../dataIn/", "../dataOut/"

filename = dataIn + "iris.csv"

data = np.loadtxt(filename, skiprows=1, delimiter=",", dtype="str")

print(f"차원수 : {data.ndim}")
# <U10 : Little Endian, Unicode String, 최대 10자리까지 표현 가능
print(f"데이터 유형 : {data.dtype}")
print(f"데이터 형상 : {data.shape}")

column_size = data.shape[1]

y_column = 1 # 정답(종속) 컬럼 개수
x_column = column_size - y_column # 독립 변수 컬럼 개수

x = data[:, 0:x_column]
# str로 타입으로 가져와서 숫자로 사용하려고 형변환함
x = x.astype(float)

y = data[:, x_column:]
# ravel() : 2차원 3차원인 numpy 배열을 1차원으로 풀어주는 함수
y = y.ravel()

# DataFrame 생성
myframe = pd.DataFrame(x, index=y)
print(myframe.head())
print()
species_index = myframe.index.unique()
print(species_index)

fig = plt.figure()
ax = fig.add_subplot(1, 1, 1)

def scatter(species, color):
    # 품종(species)별로 색상(color)을 지정하여, 산점도 그리기
    row = myframe.loc[species]
    # row[0] : 꽃받침 길이(Sepal.Length)
    # row[0] : 꽃받침 너비(Sepal.Width)
    ax.scatter(row[0], row[1], c=color, label=species)
# end def

colors = ["red", "blue", "green"]

for idx in range(len(species_index)) :
    scatter(species_index[idx], colors[idx])


ax.legend()

plt.title("꽃받침 길이와 너비의 산점도", size=19)
filename = dataOut + "iris-scatter.png"
plt.savefig(filename)


encoder = LabelEncoder()
encoder.fit(y)

y = encoder.transform(y)
print("after")
print(y[:10])



# 데이터 셋 분리
x_train, x_test, y_train, y_test = train_test_split(x, y, test_size=0.2, random_state=10)

# min과 max 그리고, round 함수는 파이썬의 내장 함수입니다.
def print_feature(somedata):
    for idx in range(somedata.shape[1]):
        # 각 컬럼별 최소 값과 최대 값 확인
        print(f'컬럼{(idx+1)} 범위 : [{round(min(somedata[:, idx]), 3)}, {round(max(somedata[:, idx]), 3)}]')

print('\n# before normalization')
print_feature(x_train)

# 거리(distance) 기반의 알고리즘인 SVM은 정규화를 진행해 주는 것이 모델 성능 향상을 기대할 수 있습니다.
from sklearn.preprocessing import StandardScaler

scaler = StandardScaler() # 표준화를 수행해주는 도구 1개
scaler.fit(x_train) # 훈련 데이터를 '기준'을 배움(평균, 퍼진 정도)

x_train = scaler.transform(x_train) # 그 기준으로 훈련 데이터 실제 변환
# x_test는 fit을 사용하지 않습니다.
# why : 학습(train) 시 사용했던 기준과 동일하게 진행해야 하므로, x_test을 사용하면 안됩니다.
x_test = scaler.transform(x_test) # 같은 기준으로 테스트 데이터도 변환

print('\n# after normalization')
print_feature(x_train)


from sklearn.svm import SVC
# model = SVC(kernel='rbf', C=8, gamma=0.1)
model = SVC(kernel='rbf', C=1, gamma=0.1)
# model = SVC(kernel='rbf', C=100, gamma=0.1)

model.fit(x_train, y_train)  # svm 분류 모델 훈련

prediction = model.predict(x_test)  # 테스트

print('예측 데이터 :', prediction)
print('실제 정답 :', y_test)

# 보기 좋게 데이터 프레임으로 변환합니다.
bool_result = prediction == y_test

result_frame = pd.DataFrame([prediction, y_test, bool_result])
print(result_frame.head())


print('\n# 데이터 프레임(예측값, 실제값, 예측 결과)')
result_frame = pd.DataFrame([prediction, y_test, bool_result])
result_frame = result_frame.T # 전치(행렬 바꿈)
result_frame.columns = ['예측값', '실제값', '예측 결과']

print('\n# result_frame')
print(result_frame)


hit_count = np.sum(bool_result)
total_count = len(result_frame.index)

print(f'\n정확도 {hit_count}/{total_count}={(hit_count/total_count):.3f}')
print(f'prediction accurary : {np.mean(bool_result):.3f}')

print('\n# 정확도 계산')
from sklearn.metrics import accuracy_score
accuracy = accuracy_score(y_test, prediction)
print(f'accuracy : {100.0 * accuracy:.3f}%')


print('\n# confusion matrix')
from sklearn.metrics import confusion_matrix
matrix = confusion_matrix(y_test, prediction)
print(matrix)

import seaborn as sns

label_list = encoder.classes_
print(label_list)

plt.figure(figsize=(10,7))

fig, ax = plt.subplots(figsize=(10,7))

sns.heatmap(
    matrix,
    annot=True,
    fmt='d',
    cmap='Blues',
    xticklabels=label_list,
    yticklabels=label_list,
    ax=ax
)

ax.set_xlabel("예측값", fontsize=15)
ax.set_ylabel("정답값", fontsize=15)
ax.set_title("Confusion Matrix", fontsize=18)

filename = dataOut + "confusion_matrix.png"
plt.savefig(filename)

plt.xlabel('예측값', size=15)
plt.ylabel('정답값', size=15)
plt.title('Confusion Matrix', size=18)

filename = dataOut + 'confusion_matrix.png'
plt.savefig(filename)


from sklearn.metrics import accuracy_score, confusion_matrix, classification_report

report = classification_report(y_test, prediction)
print(report)
```
