- 코드 내용
```python
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from sklearn.linear_model import LinearRegression
from sklearn.model_selection import train_test_split

plt.rc('font', family='Malgun Gothic')

plt.rcParams['axes.unicode_minus'] = False

pd.set_option('display.max_columns', None)

import seaborn as sns

df = pd.read_csv('../dataIn/multiple_linear_regression_dataset.csv')

print(df.head())
print()
print(df.info())
print()
print(df.describe())
print()
print(df.isnull().sum())
print()
# duplicated()는 True 혹은 False 반환함
print(df.duplicated().sum())

dataOut = "../dataOut/"

sns.pairplot(df)
plt.savefig(dataOut + "pairplot.png")

# correlation (상관계수)
corr = df.corr(numeric_only=True)
print(corr)

mask = np.triu(np.ones_like(corr, dtype=bool))
plt.figure(figsize=(10, 8))
sns.heatmap(corr, mask=mask, cmap='coolwarm', annot=True, fmt='.2f', cbar=True, square=True)
plt.title('변수들간 상관 계수', size=15)
filename = dataOut + 'heatmap.png'
plt.savefig(filename)


xdata = ['age', 'experience']
fig, axes = plt.subplots(1, 2, figsize=(14, 5))

for idx in range(len(xdata)):
    axes[idx].scatter(df[f'{xdata[idx]}'], df['income'], c='coral', s=50)
    axes[idx].set_title(f'{xdata[idx]} vs Income')
    axes[idx].set_xlabel(f'{xdata[idx]}')
    axes[idx].set_ylabel('Income')

filename = dataOut + 'scatter.png'
plt.savefig(filename)
print(filename + ' 파일이 저장되었습니다.')

x = df[["age", "experience"]]
y = df["income"]

# 10세트는 따로 빼둠
x_train, x_test, y_train, y_test = train_test_split(x, y, test_size=0.3, random_state=10)

model = LinearRegression()

model.fit(x_train, y_train)

r_square = model.score(x_test, y_test)
print(f"결정 계수 : {r_square}")
print(f"\n기울기 a : {model.coef_}")
print(f"\ny절편 b : {model.intercept_}")

y_pred = model.predict(x_test)


test_preds = pd.DataFrame(y_test)
test_preds.columns = ["정답"]
# 존재하지 않는 컬럼 이름에 값을 대입하면 새로운 컬럼이 자동으로 추가됨
test_preds["예측"] = y_pred
test_preds["mse"] = (test_preds["예측"] - test_preds["정답"]) ** 2
print("test_preds")
print(test_preds)

filename = dataOut + "test_preds.csv"
test_preds.to_csv(filename, index=False, encoding="UTF-8")

mse = test_preds["mse"].mean()
print(f"평균 제곱 오차 : {mse}")


plt.figure()

sns.regplot(x='정답', y='예측',  data=test_preds, line_kws={"color": "red"});
plt.title('정답과 예측치의 산점도')
plt.xlim([15000, 45000])
plt.ylim([15000, 45000])
plt.tight_layout()
filename = dataOut + 'regplot.png'
plt.savefig(filename)
print(filename + ' 파일이 저장되었습니다.')
```
