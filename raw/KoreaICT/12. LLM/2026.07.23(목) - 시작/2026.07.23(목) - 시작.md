```python
from dotenv import load_dotenv

# load_dotenv() # 현재 프로젝트에 있는 경우

def get_api_key(key_name: str) -> str | None:
    """
    환경 변수 값을 반환한다.
    없으면 None을 반환한다.
    """
    api_key = os.getenv(key_name)

    if api_key is None:
        print(f"\n환경 변수 '{key_name}'를 찾을 수 없습니다.")
        return None

    return api_key


# hidden_size = 50 # 보안을 위해 API Key의 뒤쪽을 *로 가리기 위한 숫자입니다.
# OpenAI API Key를 일부는 그대로 출력하고, 나머지는 *로 가리는(masking) 역할을 합니다.
def print_api_key(key_name: str = "OPENAI_API_KEY", hidden_size: int = 50):
    """API Key를 일부만 출력"""
    api_key = os.getenv(key_name)

    if not api_key:
        print(f"\n{key_name}를 찾을 수 없습니다.")
        return

    print(f"\nAPI Key를 일부만 출력")
    hidden_size = min(hidden_size, len(api_key))
    print(f"{key_name} : [{api_key[:-hidden_size]}{'*' * hidden_size}]")


def get_api_keys(key_names: tuple[str, ...]) -> dict[str, str | None]:
    """
    여러 개의 환경 변수를 한 번에 반환한다.

    Parameters
    ----------
    key_names : tuple[str, ...]
        환경 변수 이름들의 튜플

    Returns
    -------
    dict[str, str | None]
    """
    result = {}

    for key_name in key_names:
        api_key = os.getenv(key_name)

        if api_key is None:
            print(f"환경 변수 '{key_name}'를 찾을 수 없습니다.")

        result[key_name] = api_key

    return result

if __name__ == "__main__":
    print(f".env 파일의 위치 : {ENV_FILE_PATH}")

'''
일반 파일에서 다음과 같이 간단히 사용할 수 있습니다.

find_api = "OPENAI_API_KEY"
api_key = get_api_key(find_api)

client = OpenAI(api_key=api_key)
'''
```

#### \# create() 함수 : temperature (LLM(이론).pdf (P.31))
temperature는 프롬프트 엔지니어링에서 "출력 성향을 조절하는 핵심 매개 변수" 중 하나입니다.

- temperature 매개 변수 (0~2)

| 0.0 ~ 0.2                            | 0.3 ~ 0.6                         | 0.7 ~ 1.0                             | > 1.0                                            |
| ------------------------------------ | --------------------------------- | ------------------------------------- | ------------------------------------------------ |
| 거의 항상 같은 답<br>가장 안전한 표현<br>창의성 거의 없음 | 적당한 변주<br>설명은 안정적<br>약간의 표현 차이 존재 | 표현 다양<br>주제 내용 달라질 수 있음<br>말투와 비유가 풍부 | 매우 창의적<br>일관성 급감<br>사실 오류(환각) 가능성 증가<br>무작위성 극대화 |

(0.7 <= temperature <= 1)

temperature 매개 변수의 값을 변경해 가면서, 얼마나 창의적으로 답변하는 지를 확인하도록 합니다.

- 권장 예시
```text
추천 예제 1: 창의적인 이야기 생성
"content": "고양이가 운영하는 커피숍에 대한 짧은 이야기를 만들어 주세요."
temperature=0 → 비교적 일관된 이야기
temperature=2 → 매우 다양한 전개

추천 예제 2: 상품 이름 만들기
"content": "AI를 활용한 스마트 우산의 브랜드 이름을 5개 만들어 주세요."

추천 예제 3: 여행 슬로건 만들기

추천 예제 4: 가상의 국가 설정
"content": "아틀란티스라는 가상의 나라의 수도와 국기를 상상해서 설명해 주세요."
```


- 코드 내용
```python
from utility.env_util import get_api_key
from openai import OpenAI

find_api = "OPENAI_API_KEY"
api_key = get_api_key(find_api)

client = OpenAI(api_key=api_key)

def generate_response(temp):
    response = client.chat.completions.create(
        # 사용할 모델 설정
        model="gpt-4.1-nano",
        messages = [
            {
                "role":"user",
                "content":"아틀란티스라는 가상의 나라의 수도와 국기를 상상해서 설명해 주세요."
            }
        ],
        # 최대 토큰 설정
        max_completion_tokens=100,
        temperature=temp
    )

    print(f"temperature={temp}", end=f"\n{'-'*50}\n")
    print(response.choices[0].message.content, end="\n\n")

# temperature : 답변의 무작위성과 다양성을 조절
# 0에 가까울수록 일관되고 안정적인 답변을 생성
# 2에 가까울수록 답변이 다양하고 창의적이지만,
# 사실 오류나 앞뒤가 맞지 않는 내용이 나올 가능성이 커짐
for temp in [0, 1, 2]:
    generate_response(temp)
```

|     항목      | 설명                                                                                             |
| :---------: | ---------------------------------------------------------------------------------------------- |

#### \# 대화형 텍스트 생성 - 모델 응답 방식 (LLM(이론).pdf (P.44))
- 비스트리밍(Non-Streaming)
```text
기본 동작
모델이 전체 응답 생성을 완료할 때까지 클라이언트는 대기
응답이 완전히 준비되면, 서버는 모든 데이터를 단일 JSON 객체로 한 번에 클라이언트에 전송
			→ 최종 결과물이 하나의 ChatCompletion 객체로 반환
짧은 질문이나, 전체 결과가 한번에 필요한 백엔드 처리 작업에 적합

```

- 스트리밍(Streaming)
```text
모델이 텍스트를 생성하는 즉시 서버는 응답을 작은 청크(Chunk) 단위로 쪼개어 클라이언트에 지속적으로 전송
클라이언트는 청크를 실시간으로 받아 처리
첫 번째 토큰을 받는 시간이 단축되어, 사용자는 답변이 생성되는 과정을 즉시 볼 수 있어 체감 응답 속도가 개선됨
함수는 단일 객체가 아닌 이터레이터(iterator)를 반환하며, 이 이터레이터를 반복하면서 ChatCompletionChunk 객체들을 순차적으로 받아서 처리
웹 애플리케이션이나 챗봇 등 사용자에게 실시간 피드백을 제공해야 하는 인터페이스에 적합
```



#### \# 프롬프트 엔지니어링 - 프롬프트 엔지니어링(Prompt Engineering) 개요 (LLM(이론).pdf (P.50~53))
LLM이 원하는 답을 정확하고 일관되게 생성할 수 있도록 질문(프롬프트)을 설계·구성·조정하는 기술

- 프롬프트(Prompt)
```text
LLM에게 '무엇을, 어떻게 해달라'고 지시하는 입력 문장 전체를 말합니다.
사람에게 질문할 때도 요청을 명확히 해야 원하는 답을 얻을 수 있듯, LLM에게도 프롬프트가 곧 "명령어이자 대화의 시작점"이 됩니다.
즉, AI에게 말해 주는 작업 설명서입니다
```

- 사용 예시
```text
예시01 : 오늘 서울 날씨 알려줘
예시02 : 서울의 오늘 날씨를 초등학생도 이해할 수 있게 3문장으로 설명해 줘
```

```text

Q: 신입 개발자를 위하여 리눅스 파일 시스템을 5문장 이내로 예시와 함께 설명해줘

```

#### \# 프롬프트 엔지니어링 - 프롬프트 엔지니어링(Prompt Engineering) 개요 (LLM(이론).pdf (P.54~55))
- 프롬프트 종류 (role에 들어가는 내용 : system, user, assistant)

| 종류 | 역할 | 예시 |
|---|---|---|
| System Prompt | 모델의 성격과 역할을 설정 | "너는 친절하고 이해심 많은 선생님이야." |
| User Prompt | 사용자의 실제 질문이나 요청 | "인공지능이 뭐야?" |
| Assistant Prompt | 모델이 이전에 했던 응답 | "인공지능은 사람의 학습 능력을 모방한 기술이야." |
```text
#### 왜 이렇게 구분할까?
- System Prompt → 모델의 기본 성격을 결정합니다.(친절한 선생님 / 냉정한 분석가 등)
- User Prompt → 사용자가 무엇을 알고 싶은지를 알려줍니다.
- Assistant Prompt → 이전 대화 내용을 기억하고 맥락을 유지하게 도와줍니다.
  
이 세 가지를 함께 쓰면 모델은
> “지금 나는 친절한 선생님이고, 방금 "인공지능이 뭐냐"는 질문을 들었구나."
라고 이해한 뒤 더 자연스럽고 일관된 답변을 합니다.
```


- 구성 요소

| 구성 요소 | 역할 | 설명 |
|---|---|---|
| System 메시지 | AI 모델의 성격과 행동 규칙 정의 | AI에게 역할/원칙/지시 사항 제시<br>예시) "친절한 초등학교 교사처럼 설명하라", "항상 한국어로 답변하라" |
| User 메시지 | AI가 수행해야 할 작업 요청 | 사용자의 질문, 요구 사항을 전달 |
| Context | 참고 자료 및 배경 정보 제공 | 정확하고 관련성 높은 답변을 생성할 수 있도록 추가 정보, 배경 지식, 예시, 문서, 데이터 등 제공 |
| Constraints | 출력 형식 및 범위 제한 | 출력 제한 사항 : 답변의 길이, 형식, 톤, 언어, 금지 사항 등을 지정 |

```python
def multi_turn_chat():

            print_message_history(messages)
            break

        messages.append({

                "content": user_input
        })

        # OpenAI에 요청함
        answer = get_ai_response(messages)

        # AI가 응답한 messages도 누적 시킵니다.
        messages.append({
                "role": "assistant",
                "content": answer
        })

        print("AI : ", answer)

def print_message_history(total_msg):
    print("\n" + "=" * 60)

        role = msg["role"]

        if role == "system":
            speaker = "System"
        elif role == "user":
            speaker = "User"
        else:
            speaker = "Assistant"

        print(f"[{i}] {speaker}")
        print(msg["content"])
        print("-" * 60)
```
