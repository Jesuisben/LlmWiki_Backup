```python
import pandas as pd
from faker import Faker

fake = Faker('ko_KR') # 페이커 객체
dataIn = "../dataIn/"

result01 = []
columns01 = ('아이디', '이름', '주소', '전화번호', '이메일')
myencoding = 'CP949' # 'CP949'  'UTF-8'  인코딩 문자열
```

# 실제로 fake (Faker 객체)에 데이터가 없어서
# 원래는 fake.user_name()같은게 불가능한데
# Faker는 가짜 데이터 (더미 데이터)여서 가능함
for idx in range(10):
    sublist = []
    sublist.append(fake.user_name())
    sublist.append(fake.name())
    sublist.append(fake.address())
    sublist.append(fake.phone_number())
    sublist.append(fake.email())
    result01.append(sublist)

# 데이터 프레임 내용 확인
myframe01 = pd.DataFrame(result01, columns = columns01)
print('데이터 프레임 내용 확인')
myframe01

filename = dataIn + "result01_csv"
# csv : 만만한 텍스트 파일
# to_csv() : to를 붙이면 csv파일로 만든다는 뜻임
myframe01.to_csv(filename , encoding=myencoding, mode='w', index=True)
